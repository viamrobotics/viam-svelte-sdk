---
title: useMachineStatus
description: Reactive query for the machine's status, deduplicated at the ViamProvider level.
---

Wraps `robotClient.getMachineStatus()` in a reactive query at the [`ViamProvider`](../../guides/getting-started/) level.

```svelte
<script lang="ts">
import { useMachineStatus } from '@viamrobotics/svelte-sdk';

interface Props {
  partID: string;
}

let { partID }: Props = $props();

const machineStatus = useMachineStatus(() => partID);

$inspect(machineStatus.current);
$inspect(machineStatus.query.error);
$inspect(machineStatus.query.isFetching);
</script>
```
